\section{大括号排版}
$\backslash$left加边界符结合$\backslash$array或$\backslash$align命令.
\begin{displaymath}
F^{HLLC}=\left\{
\begin{array}{rcl}
F_L   & & (0 < S_L)\\
F^*_L & & (S_L \leq 0 < S_H)\\
F^*_R & & (S_M \leq 0 < S_R)\\
F_R   & & (S_R \leq 0)
\end{array}
\right.
\end{displaymath}
使用$\backslash$cases命令.
\begin{displaymath}
F^{HLLC}=
\begin{cases}
F_L   & (0 < S_L)\\
F^*_L & (S_L \leq 0 < S_H)\\
F^*_R & (S_M \leq 0 < S_R)\\
F_R   & (S_R \leq 0)
\end{cases}
\end{displaymath}
